# Epic Link & Group-by-Parent Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let the user view, set, and group by an issue's parent epic in ripjira — with config-driven recognition of custom epic-shaped issuetypes (e.g. `Epic Feature`).

**Architecture:** Extend the domain `Issue` with `ParentKey`/`ParentSummary`, fed by `fields.parent` in the existing `search/jql` pipeline. Add `Client.SetParent` (PUT) and `Client.SearchEpics` (JQL by configurable issuetype list). New `EpicPicker` overlay opens on `E`, dispatches an optimistic `SetParent`. New `ByParent` grouping strategy buckets tasks under their parent. One config key `epic_issue_types` defaults to `["Epic", "Epic Feature"]` and feeds both the grouping strategy and the picker JQL.

**Tech Stack:** Go 1.26, Bubble Tea / lipgloss, `gopkg.in/yaml.v3` (already vendored), `httptest` for HTTP-level tests, `charmbracelet/x/exp/teatest` for the integration check.

**Spec:** `docs/superpowers/specs/2026-05-04-epic-link-design.md`.

```go
// SetParent attaches issue key to parentKey via PUT /issue/{key} on the
// standard parent field. Pass parentKey == "" to detach the issue from its
// current parent (Jira interprets a JSON null as "remove parent"). The
// caller should already have read access to parentKey; Jira validates the
// link server-side.
func (c *Client) SetParent(ctx context.Context, key, parentKey string) error {
	if key == "" {
		return errors.New("jira: issue key is required")
	}
	var parent any // nil → JSON null → detach
	if parentKey != "" {
		parent = map[string]string{"key": parentKey}
	}
	return c.UpdateIssue(ctx, key, map[string]any{"parent": parent})
}
```

Note: `UpdateIssue` rejects empty `fields`, but `{"parent": nil}` is one
non-empty entry (the value being nil is fine — Go marshals nil to `null`).

- [ ] **Step 5: Implement `SearchEpics`**

Append to `internal/jira/issues.go`:

```go
// SearchEpics returns issues in projectKey whose issuetype name matches any
// of epicTypes (sent literally as a JQL `IN (...)` clause), ordered by
// updated DESC. Used by the epic picker; result is the unbounded Search()
// page, but in practice projects have well under 1000 epics.
func (c *Client) SearchEpics(ctx context.Context, projectKey string, epicTypes []string) ([]Issue, error) {
	if projectKey == "" {
		return nil, errors.New("jira: project key is required")
	}
	if len(epicTypes) == 0 {
		return nil, errors.New("jira: epic types list is empty")
	}
	jql := buildEpicJQL(projectKey, epicTypes)
	return c.Search(ctx, jql)
}

// buildEpicJQL composes the JQL for SearchEpics. Project key and each epic
// type are wrapped in double quotes; embedded quotes/backslashes are
// backslash-escaped so user-supplied issuetype names like `My "Epic"` round
// trip safely.
func buildEpicJQL(projectKey string, epicTypes []string) string {
	q := func(s string) string {
		s = strings.ReplaceAll(s, `\`, `\\`)
		s = strings.ReplaceAll(s, `"`, `\"`)
		return `"` + s + `"`
	}
	parts := make([]string, len(epicTypes))
	for i, t := range epicTypes {
		parts[i] = q(t)
	}
	return fmt.Sprintf(`project = %s AND issuetype IN (%s) ORDER BY updated DESC`,
		q(projectKey), strings.Join(parts, ", "))
}
```

Add `"fmt"` to the imports if not already present.

Append to `internal/tui/grouping/grouping.go`:

```go
// ByParent groups issues into three zones:
//
//  1. "Epics" at the top — issues whose Type.Name (case-insensitive)
//     matches any entry in EpicTypes; sorted by priority desc, updated desc.
//  2. One bucket per non-empty ParentKey for the remaining issues. Bucket
//     key is "KEY  Summary" when ParentSummary is known, else just "KEY".
//     Buckets are ordered alphabetically by parent key.
//  3. "No epic" at the bottom — issues with empty ParentKey (and not epic).
//
// Inside each non-Epic bucket: priority desc, updated desc.
type ByParent struct {
	EpicTypes []string
}

// Name implements Strategy.
func (ByParent) Name() string { return "parent" }

// Group implements Strategy.
func (b ByParent) Group(issues []jira.Issue) []Group {
	isEpic := map[string]bool{}
	for _, t := range b.EpicTypes {
		isEpic[strings.ToLower(t)] = true
	}

	var epics, orphans []jira.Issue
	parentKeys := []string{}
	parentLabel := map[string]string{}
	parentBuckets := map[string][]jira.Issue{}

	for _, is := range issues {
		if isEpic[strings.ToLower(is.Type.Name)] {
			epics = append(epics, is)
			continue
		}
		if is.ParentKey == "" {
			orphans = append(orphans, is)
			continue
		}
		if _, seen := parentBuckets[is.ParentKey]; !seen {
			parentKeys = append(parentKeys, is.ParentKey)
			label := is.ParentKey
			if is.ParentSummary != "" {
				label = is.ParentKey + "  " + is.ParentSummary
			}
			parentLabel[is.ParentKey] = label
		}
		parentBuckets[is.ParentKey] = append(parentBuckets[is.ParentKey], is)
	}

	sort.Strings(parentKeys)
	sortByPriorityDesc(epics)
	sortByPriorityDesc(orphans)
	for k := range parentBuckets {
		v := parentBuckets[k]
		sortByPriorityDesc(v)
		parentBuckets[k] = v
	}

	out := make([]Group, 0, 2+len(parentKeys))
	if len(epics) > 0 {
		out = append(out, Group{Key: "Epics", Issues: epics})
	}
	for _, k := range parentKeys {
		// Refresh the label from the first issue in the bucket — its
		// ParentSummary is more likely to be filled than later ones if the
		// data is uneven. (Bucket is non-empty by construction.)
		label := parentLabel[k]
		if first := parentBuckets[k][0]; first.ParentSummary != "" {
			label = k + "  " + first.ParentSummary
		}
		out = append(out, Group{Key: label, Issues: parentBuckets[k]})
	}
	if len(orphans) > 0 {
		out = append(out, Group{Key: "No epic", Issues: orphans})
	}
	return out
}
```

// ByName returns the strategy registered under name (case-insensitive).
// epicTypes feeds ByParent; pass nil for any caller not using "parent".
// Empty or unknown names fall back to ByStatus, matching the config default.
func ByName(name string, epicTypes []string) Strategy {
	switch strings.ToLower(strings.TrimSpace(name)) {
	case "priority":
		return ByPriority{}
	case "epic":
		return ByEpicAndPriority{}
	case "parent":
		return ByParent{EpicTypes: epicTypes}
	}
	return ByStatus{}
}

```go
package overlays

import (
	"strings"

	"github.com/billygate/ripjira/internal/jira"
	"github.com/billygate/ripjira/internal/tui/styles"
	"github.com/charmbracelet/bubbles/key"
	tea "github.com/charmbracelet/bubbletea"
	"github.com/charmbracelet/lipgloss"
)

// EpicPickedMsg is emitted when the user confirms a choice. ParentKey is
// empty for the "detach" row.
type EpicPickedMsg struct {
	IssueKey  string
	ParentKey string
}

// EpicCancelledMsg is emitted on Esc.
type EpicCancelledMsg struct{}

// Epic is the `E` overlay: pick an epic for the focused issue. j/k or
// arrows navigate; runes type into a fuzzy filter; Enter picks; Esc
// cancels. Show before SetEpics: SetEpics is the async-loaded result of
// SearchEpics and arrives later via a typed message.
type Epic struct {
	visible       bool
	issueKey      string
	currentParent string
	all           []jira.Issue
	filter        string
	cursor        int
}

// NewEpic constructs a hidden picker.
func NewEpic() Epic { return Epic{} }

// Visible reports whether the overlay is shown.
func (e Epic) Visible() bool { return e.visible }

// IssueKey returns the issue the overlay was opened for.
func (e Epic) IssueKey() string { return e.issueKey }

// Show opens the picker; current is the issue's existing ParentKey (or "").
func (e Epic) Show(issueKey, current string) Epic {
	e.visible = true
	e.issueKey = issueKey
	e.currentParent = current
	e.cursor = 0
	e.filter = ""
	return e
}

// Hide returns a copy of e with state cleared.
func (e Epic) Hide() Epic {
	return Epic{}
}

// SetEpics replaces the candidate epic list (typically the result of
// jira.Client.SearchEpics).
func (e Epic) SetEpics(all []jira.Issue) Epic {
	e.all = append([]jira.Issue(nil), all...)
	if e.cursor >= e.totalRows() {
		e.cursor = 0
	}
	return e
}

// detachRowVisible reports whether the leading "No epic" row is shown.
func (e Epic) detachRowVisible() bool { return e.currentParent != "" }

// rows returns the visible epic items after filter; index 0 is the detach
// row when present.
func (e Epic) rows() []jira.Issue {
	if e.filter == "" {
		return e.all
	}
	needle := strings.ToLower(e.filter)
	out := make([]jira.Issue, 0, len(e.all))
	for _, ep := range e.all {
		hay := strings.ToLower(ep.Key + " " + ep.Summary)
		if strings.Contains(hay, needle) {
			out = append(out, ep)
		}
	}
	return out
}

func (e Epic) totalRows() int {
	n := len(e.rows())
	if e.detachRowVisible() {
		n++
	}
	return n
}

// Update consumes input while visible.
func (e Epic) Update(msg tea.Msg) (Epic, tea.Cmd) {
	if !e.visible {
		return e, nil
	}
	k, ok := msg.(tea.KeyMsg)
	if !ok {
		return e, nil
	}
	switch {
	case key.Matches(k, key.NewBinding(key.WithKeys("up", "k"))):
		if e.cursor > 0 {
			e.cursor--
		}
	case key.Matches(k, key.NewBinding(key.WithKeys("down", "j"))):
		if e.cursor < e.totalRows()-1 {
			e.cursor++
		}
	case key.Matches(k, key.NewBinding(key.WithKeys("backspace"))):
		if len(e.filter) > 0 {
			e.filter = e.filter[:len(e.filter)-1]
			if e.cursor >= e.totalRows() {
				e.cursor = 0
			}
		}
	case key.Matches(k, key.NewBinding(key.WithKeys("enter"))):
		issueKey := e.issueKey
		parent := e.parentKeyAtCursor()
		e.visible = false
		return e, func() tea.Msg { return EpicPickedMsg{IssueKey: issueKey, ParentKey: parent} }
	case key.Matches(k, key.NewBinding(key.WithKeys("esc"))):
		e.visible = false
		return e, func() tea.Msg { return EpicCancelledMsg{} }
	default:
		if k.Type == tea.KeyRunes {
			e.filter += string(k.Runes)
			e.cursor = 0
		}
	}
	return e, nil
}

func (e Epic) parentKeyAtCursor() string {
	if e.detachRowVisible() {
		if e.cursor == 0 {
			return ""
		}
		idx := e.cursor - 1
		rows := e.rows()
		if idx >= 0 && idx < len(rows) {
			return rows[idx].Key
		}
		return ""
	}
	rows := e.rows()
	if e.cursor < 0 || e.cursor >= len(rows) {
		return ""
	}
	return rows[e.cursor].Key
}

// View renders the overlay. The styles parameter is the project's shared
// Styles struct; only Border/Padding-shaped fields are referenced.
func (e Epic) View(s styles.Styles) string {
	if !e.visible {
		return ""
	}
	var lines []string
	header := "Set epic for " + e.issueKey
	if e.filter != "" {
		header += "  /" + e.filter
	}
	lines = append(lines, header)

	row := func(label string, selected bool) string {
		marker := "  "
		if selected {
			marker = "▸ "
		}
		return marker + label
	}

	idx := 0
	if e.detachRowVisible() {
		lines = append(lines, row("⊘ No epic (detach)", e.cursor == idx))
		idx++
	}
	rows := e.rows()
	if len(rows) == 0 && !e.detachRowVisible() {
		lines = append(lines, "  (no epics)")
	}
	for _, r := range rows {
		label := r.Key
		if r.Summary != "" {
			label = r.Key + "  " + r.Summary
		}
		lines = append(lines, row(label, e.cursor == idx))
		idx++
	}
	lines = append(lines, "↑/↓ navigate  type to filter  ⏎ select  esc cancel")

	return lipgloss.NewStyle().
		Border(lipgloss.RoundedBorder()).
		Padding(0, 1).
		Render(lipgloss.JoinVertical(lipgloss.Left, lines...))
}
```
